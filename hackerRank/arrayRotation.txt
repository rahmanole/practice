static int[] rotLeft(int[] a, int d) {
        int[] res = new int[a.length];
        int x = a.length - d;
        for(int i = 0;i<d;i++){
            res[x]=a[i];
            x++;
        }
        for(int j = 0;j<a.length-d;j++){
            res[j] = a[d];
            d++;
        }
        return res;
    }
